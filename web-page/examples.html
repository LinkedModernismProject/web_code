Example Searches

select * where { ?Subject ?Predicate ?Object } limit 10 our normal search
SELECT distinct ?Subject ?Predicate ?Object WHERE {?Subject ?Predicate ?Object } ORDER BY DESC(?Subject) LIMIT 20 Good ex.
SELECT distinct * WHERE { ?Subject ?Predicate ?Object } ORDER BY RAND() LIMIT 2 Gets 2 kinds of ?Predicate
SELECT distinct * WHERE { ?Subject ?Predicate ?Object FILTER regex(?Subject, "Jean")} ORDER BY ASC(?Object) LIMIT 5 Filters
SELECT distinct * WHERE { ?Subject foaf:name ?name. ?Subject ?Predicate ?Object } limit 24
PREFIX ty: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> SELECT * WHERE {{ ?Predicate a ty:Property} } limit 10
SELECT distinct * WHERE{ ?node foaf:givenName ?givenName. ?node foaf:name ?name . ?node <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> <http://xmlns.com/foaf/0.1/Person> } LIMIT 10
